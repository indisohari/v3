const SPREADSHEET_ID = "여기에_스프레드시트_ID를_넣으세요"; const SHEET_NAME = "responses"; /** * 선택: 브라우저에서 열었을 때 상태 확인용 */ function doGet() { return jsonOutput_({ success: true, message: "RSVP endpoint is running", time: new Date().toISOString() }); } /** * RSVP 저장 * 프론트에서 text/plain(JSON 문자열)로 전송 */ function doPost(e) { try { if (!e || !e.postData || !e.postData.contents) { return jsonOutput_({ success: false, message: "요청 본문이 비어 있습니다." }); } const raw = e.postData.contents; let payload; try { payload = JSON.parse(raw); } catch (parseErr) { return jsonOutput_({ success: false, message: "JSON 파싱 실패" }); } const name = String(payload.name || "").trim(); const phone = String(payload.phone || "").trim(); const side = String(payload.side || "").trim(); const guestCount = Number(payload.guestCount || 0); const guestCountLabel = String(payload.guestCountLabel || "").trim(); const companionNames = String(payload.companionNames || "").trim(); // 서버측 검증 (프론트 검증만 믿지 않기) if (!name) return jsonOutput_({ success: false, message: "성함이 필요합니다." }); if (!/^01[016789]-\d{3,4}-\d{4}$/.test(phone)) { return jsonOutput_({ success: false, message: "연락처 형식이 올바르지 않습니다." }); } if (!(side === "신랑측" || side === "신부측")) { return jsonOutput_({ success: false, message: "참석 구분이 올바르지 않습니다." }); } if (!guestCount || guestCount < 1) { return jsonOutput_({ success: false, message: "참석 인원이 올바르지 않습니다." }); } const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { return jsonOutput_({ success: false, message: "지정한 시트를 찾을 수 없습니다." }); } // 첫 행 헤더 없으면 자동 생성 (선택) if (sheet.getLastRow() === 0) { sheet.appendRow([ "제출시각", "성함", "휴대폰", "참석구분", "참석인원(숫자)", "참석인원라벨", "동반인성함", "UserAgent" ]); } const userAgent = (e.parameter && e.parameter.ua) ? e.parameter.ua : ""; // 기본 빈값 const timestamp = Utilities.formatDate(new Date(), "Asia/Seoul", "yyyy-MM-dd HH:mm:ss"); sheet.appendRow([ timestamp, name, phone, side, guestCount, guestCountLabel, companionNames, userAgent ]); return jsonOutput_({ success: true, message: "저장 완료" }); } catch (err) { return jsonOutput_({ success: false, message: "서버 오류", error: String(err) }); } } /** * JSON 응답 헬퍼 */ function jsonOutput_(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
